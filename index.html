<script>
const apiKey = ""; // Gemini API Key goes here
const chatWidget = document.getElementById('chat-widget');
const chatMessages = document.getElementById('chat-messages');
const userInput = document.getElementById('user-input');
const loader = document.getElementById('loader');
function toggleChat() {
const isFlex = chatWidget.style.display === 'flex';
chatWidget.style.display = isFlex ? 'none' : 'flex';
}
async function sendMessage() {
const text = userInput.value.trim();
if (!text) return;
// User message
const userDiv = document.createElement('div');
userDiv.className = 'user-bubble';
userDiv.innerText = text;
chatMessages.appendChild(userDiv);

userInput.value = "";
loader.classList.remove('hidden');
chatMessages.scrollTop = chatMessages.scrollHeight;
try {
const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify({
contents: [{ parts: [{ text: text }] }],
systemInstruction: { parts: [{ text: "Apnar nam Bulbuli Pro. Apni ekjon premium friendly AI bondhu. Uttar sob somoy Banglish-e hobe. Emoji use korben." }] }
})
});
const data = await response.json();
const reply = data.candidates?.[0]?.content?.parts?.[0]?.text || "Ektu jhamela hocche bondhu!";
// AI Response
const aiContainer = document.createElement('div');
aiContainer.className = 'ai-msg-container';
aiContainer.innerHTML = `
<div class="ai-logo-mini">
<svg width="12" height="12" viewBox="0 0 24 24" fill="black"><path d="M12 2L14.5 9L22 11.5L14.5 14L12 22L9.5 14L2 11.5L9.5 9L12 2Z"/></svg>
</div>
<div class="ai-bubble">${reply}</div>
`;
chatMessages.appendChild(aiContainer);

} catch (e) {
const errorDiv = document.createElement('div');
errorDiv.className = 'ai-bubble';
errorDiv.innerText = "Connection error bondhu! ✨";
chatMessages.appendChild(errorDiv);
} finally {
loader.classList.add('hidden');
chatMessages.scrollTop = chatMessages.scrollHeight;
}
}
</script>
